// Add stop research function (add after line 163)
const stopResearch = async () => {
  if (!sessionId) return;
  
  try {
    await api.post(`/research/sessions/${sessionId}/stop`);
    await loadSession(); // Reload to show cancelled status
  } catch (error) {
    console.error("Error stopping research:", error);
  }
};

// Add download PDF function (add after stopResearch)
const downloadPDF = async () => {
  if (!sessionId) return;
  
  try {
    const response = await api.get(`/reports/session/${sessionId}`);
    if (response.data && response.data.download_url) {
      // Open download URL in new tab
      const downloadUrl = `http://localhost:8000${response.data.download_url}`;
      window.open(downloadUrl, '_blank');
    }
  } catch (error) {
    console.error("Error downloading PDF:", error);
    alert("PDF report not yet generated. Please wait for research to complete.");
  }
};
